def mean(*args):
    total = 0.0
    count = 0  
    for i in args:
        if type(i) in (int, float):
            total += i
            count += 1
    if count == 0:
        return 0.0
    else:
        return total / count

def fact(n):
    res = 1
    for i in range(1, n + 1):
        res *= i
    return res
